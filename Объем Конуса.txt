#include <iostream>
#include <cmath>
using namespace std;

float volume(float radius, float high)
{
	float pi = 3.14;
	float V = pi / 3 * pow(radius, 2) * high;
	return V;
}


void main()
{
	setlocale(LC_ALL, "ru");

	float radius = 2, high = 9;

	cout << "Формула - V = 1/3 * pi * r * r * h \nРадиус - 2 \nВысота - 9" << endl;
	cout <<"Объем - "<< volume(radius, high) <<endl;
	
	system("pause");
}
