Предприниматель, начав дело, взял кредит размером k
рублей под p процентов годовых и вложил его в свое дело. По прогнозам, его
дело должно давать прибыль r рублей в год. Сможет ли он накопить сумму,
достаточную для погашения кредита, и если да, то через сколько лет. Все
параметры вводятся с клавиатуры

#include <iostream>

using namespace std;

void end()
{
	cout << "\nERROR\n";
	system("pause");
	exit(0);
}

void main()
{
	setlocale(LC_ALL, "Rus");
	float k = 0, p = 0, r = 0, rfixed = 0, year = 2, res = 0;
	cout << "Введите кредит проценты и зарабаток в год: ";
	cin >> k >> p >> r;
	rfixed = r;
	p *= 0.01;
	k += k * p;

	if (k < 0 || r < 0 || p < 0)
		end();
	for (;; k += k*p)
	{
		r += rfixed;
		if (k <= r)
		{
			res++;
			break;
		}
		if (k*p > r)
		{
			res += 2;
			break;
		}
		year++;
	}
	if (res == 1)
		cout << "Выплатит за " << year << "y\n" << endl;
	if (res == 2)
		cout << "Не выплатит\n" << endl;
	cout << year << endl;

	system("pause");
}
