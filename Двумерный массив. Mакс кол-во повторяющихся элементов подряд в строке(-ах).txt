#include <iostream>
#include <ctime>

using namespace std;

void main()
{
	srand(time(NULL));
	setlocale(LC_ALL, "Rus");

	const int N = 5, M = 6;
	int Mas[N][M], strikel[N], strike = 1, maksstrikel = 0;
	for (int i = 0; i < N; i++)
	{
		for (int j = 0; j < M; j++)
		{
			Mas[i][j] = rand() % 6;
			cout << Mas[i][j] << " ";
		}
		cout << endl;
	}

	for (int i = 0; i < N; i++)
	{
		strikel[i] = 0;
		for (int j = 0; j < M - 1; j++)
		{
			if (Mas[i][j] == Mas[i][j + 1])
				strike++;
			if (strike > strikel[i])
				strikel[i] = strike;
			if (Mas[i][j] != Mas[i][j + 1])
				strike = 1;
		}
		strike = 1;
	}
	for (int i = 0; i < N - 1; i++)
	{
		if (strikel[maksstrikel] < strikel[i + 1])
			maksstrikel = i + 1;
	}
	cout << "\n Макс кол-во повторяющихся элементов подряд в строке(-ах): " << endl;
		for (int i = 0; i < N; i++)
		{
			if (strikel[maksstrikel] == strikel[i])
				cout << i + 1 << endl;
		}
	system("pause");
}
