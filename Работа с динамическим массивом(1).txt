#include <iostream>
#include<ctime>
using namespace std;

void end()
{
	cout << "\nERROR\n";
	system("pause");
	exit(0);
}

int main()
{
	setlocale(LC_ALL, "Rus");
	srand(time(NULL));

	int n = 0;

	cout << " акого размера массив вы хотите создать?" << endl;
	cin >> n;
	cout << endl;

	if (n < 1)
		end();

	float *Mas = new float[n];

	for (int i = 0; i < n; i++)
	{
		Mas[i] = (rand() % 101 - 50) / 10.0;
		cout << Mas[i] << " ";
	}

	cout << endl;

	for (int i = 0; i < n - 1; i++)
		if (Mas[i] < 0)
		{
			Mas[i] = 0;
			for (int j = i; j < n - 1; j++)
			{
				swap(Mas[j], Mas[j + 1]);
			}
			i = 0;
		}

	for (int i = 0; i < n; i++)
	{
		cout << Mas[i] << " ";
	}

	cout << endl << endl;

	delete Mas;

	system("pause");
}
