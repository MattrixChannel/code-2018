#include <iostream>
using namespace std;

void end()
{
	cout << "\nERROR\n";
	system("pause");
	exit(0);
}

int main()
{
	setlocale(LC_ALL, "ru");

	const int N = 10; // Задаем константу и переменные 
	int Mas[N], sum = 0;

	cout << "Введите баллы от 1 до 10 для 10 занятий:" << endl;

	for (int i = 0; i < N; i++) // Ввод баллов
	{
		cin >> Mas[i];
		if (Mas[i] < 1 || Mas[i] > 10) //Проверка введенного числа
		end();

		sum += Mas[i];
	}

	cout << "Баллы за 10 занятий:" << endl;

	for (int i = 0; i < N; i++) // Ввывод всех баллов за все занятия
	{
		cout << Mas[i] << " ";
	}

	cout << "\nВсего баллов >> " << sum << endl; // Ввывод суммы всех баллов

	system("pause");
}
