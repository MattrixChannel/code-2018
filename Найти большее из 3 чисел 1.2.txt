#include <iostream>
using namespace std;

void main()
{
	setlocale(LC_ALL, "ru");

	float x, y, z;

	cout << "\nВведите 3 числа" << endl;
	cout << "\n Введите 1-ое число >> ";
	cin >> x;

	cout << " Введите 2-ое число >> ";
	cin >> y;

	cout << " Введите 3-ье число >> ";
	cin >> z;

	if (x == y && x == z) //Если 3 числа равны
	{

		cout <<"  "<<"Все числа равны" << endl;

	}
	else
	{
		if (x == y && x != z && x > z) //Если 2 числа больше 3
		{

			cout << "  " << x << " и " << y << " - самые большие числа" << endl;

		}
		
		if (x == z && y != z && x > y)
		{

			cout << "  " << x << " и " << z << " - самые большие числа" << endl;

		}
		
		if (z == y && x != z && z > x)
		{

			cout << "  " << z << " и " << y << " - самые большие числа" << endl;

		}
		if (x == y && x != z && x < z) // Если 2 числа меньше 3
		{

			cout << "  " << z << " - самое большое число" << endl;

		}

		if (x == z && y != z && x < y)
		{

			cout << "  " << y << " - самое большое число" << endl;

		}

		if (z == y && x != z && z < x)
		{

			cout << "  " << x << " - самое большое число" << endl;

		}
		if (x != y && x != z && z != y) //Если все 3 числа разные
		{

			if (x > z && x > y)
			{
				cout << "  " << x << " - самое большое число" << endl;
			}

			if (y > z && y > x)
			{
				cout << "  " << y << " - самое большое число" << endl;
			}

			if (z > x && z > y)
			{
				cout << "  " << z << " - самое большое число" << endl;
			}
		}

	}
	cout << "\n";

	system("pause");
}
