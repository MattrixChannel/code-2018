#include <iostream>
#include<ctime>
using namespace std;

int main()
{
	setlocale(LC_ALL, "Rus");
	srand(time(NULL));

	const int N = 8, M = 10;
	int Mas[N][M], makszeros[N], maksz = 0;
	for (int l = 0; l < N; l++)
		makszeros[l] = 0;

	for (int j = 0; j < N; j++)
	{
		for (int i = 0; i < M; i++)
		{
			Mas[j][i] = rand() % 2;
			cout << Mas[j][i] << " ";
			if (Mas[j][i] == 0)
				makszeros[j]++;
		}
		cout << endl;
	}

	for (int j = 0; j < N; j++)
	{
		if (maksz < makszeros[j])
		{
			maksz = makszeros[j];
		}
	}

	cout << "\nВ" << endl;

	for (int i = 0; i < N; i++)
	{
		if (maksz == makszeros[i])
		cout << i + 1 << endl;
	}
	cout << "Строчкe(-ах) больше всего нулей" << endl;

	system("pause");
}
