#include <iostream>
#include<ctime>
using namespace std;

int main()
{
	srand(time(NULL));
	setlocale(LC_ALL, "ru");

	const int N = 10;
	int Mas[N];

	cout << " Ряд чисел:" << endl;

	for (int i = 0; i < N; i++)
	{
		Mas[i] = rand() % 200 - 100;
		cout << " " << Mas[i];
	}
	cout << "\n\n Противоположные числа:" << endl;

	for (int i = 0; i < N; i++)
	{
		cout << " " << -Mas[i];
	}
	cout << endl << endl;
	system("pause");
}
